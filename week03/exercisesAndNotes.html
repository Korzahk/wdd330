  • Create a calculator
    My code:
    let calculator = {
    let v1, v2;
    read() = {}
    sum() = { return this.v1 + this.v2; }
    mul() = { return this.v1 * this.v2 }
    };
    Solution:
    let calculator = {
    sum() { return this.a + this.b; },
    mul() { return this.a * this.b; },
    read() {
    this.a = +prompt('a?', 0);
    this.b = +prompt('b?', 0);
    }
    };
